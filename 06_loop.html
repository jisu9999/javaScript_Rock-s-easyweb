Loop

for 반복문

        for(초기문; 조건문; 증감문){
            //반복할 구문
        }

while 반복문

        초기문;
        while(조건문){
            // 반복될 구문
            증감문;
        }

do while 반복문

      초기문;
      do{
        // 반복될 구문
        증감문;
      } while(조건문);

forEach 반복문

      대상 - 배열
      대상.forEach(function(각원소를 대변할 매개변수, 각원소의 인덱스, 전체원소){
           //반복할 일
      });

for in 반복문

객체에서만 쓸 수 있음! 객체의 값에서 반복문 출력

      for(각 프로퍼티 변수 in 객체명){
          반복할 일;
      }

for of 반복문

배열에서 사용할 수 있는 반복문(forEach랑 비슷)
배열의 값 반복문

      for(각원소의 변수명 of 배열명){
          반복할 일
      }
